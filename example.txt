func fibonacci(enteredNumber int) int {    
    var result int    
    if enteredNumber == 1 || enteredNumber == 2 {    
        result = (enteredNumber - 1)    
    } else {    
        result = fibonacci(enteredNumber - 1) + fibonacci(enteredNumber - 2)    
    }    
    return result    
}
